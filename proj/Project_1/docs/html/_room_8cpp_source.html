121 void Flow::Room::trigger(){
122  bool moved = false; //If the player has moved
123  std::stringstream convert; //String stream for conversion
124  convert << "GameData/desc" << (static_cast<int>(_event) * 3) + ((Game::gmRand.rand() % 3) + 1) << ".txt"; //Get description path
125  rdTxt(convert.str()); //Display Room description
126  std::cout << std::endl;
127 
128  if(_event != RmEvent::None){ //If the RmEvent is not none
129  if(_event == RmEvent::Encounter){ //If encounter
130  Actor enem = Game::gmRand.rActor(); //Create enemy
131  std::cout << "A " << enem.name() << " attacks!" << std::endl;
132  Game::over = !Flow::encounter(enem); //Process encounter
133  }
134  else if(_event == RmEvent::Treasure){ //If treasure
135  unsigned char tCount = (Game::gmRand.rand() % 3) + 1; //Generate between 1 and 3 treasures
136 
137  for(int i = 0; i < tCount; ++i){ //For each treasure
138  Item treas = Game::gmRand.rItem(); //Generate a treasure
139  std::cout << treas.uiName() << " acquired!" << std::endl;
140  Game::player.addItem(treas); //Add treasure to the player's inventory
141  }
142  }
143  else if(_event == RmEvent::Spring){ //If spring
144  Game::player.setHp(Game::player.hp().max()); //Heal player
145  Game::player.setMp(Game::player.mp().max()); //Heal player
146  std::cout << Game::player.name() << " is fully restored!" << std::endl;
147  for(int i = 0; i < Game::player.invSize(); ++i){ //Identify Items
148  Game::player.identify(i); //Identify
149  }
150  std::cout << "Items identified!" << std::endl;
151  }
152  _event = RmEvent::None; //Clear room event
153  }
154  else{ //If the event is None
155  std::cout << "There are exits to the ( " << Direct::toStr(_exit) << ")" << std::endl; //Display Exits
156  if(_end){ //If this is the end Room
157  std::cout << "You can see a flight of stairs in the far corner!" << std::endl; //Display End message
158  }
159  std::vector<std::string> tmpMenu(Game::gMenu); //Add exits to game menu
160  if(FlgUtil::hasFlag(_exit, Direct::NORTH)){
161  tmpMenu.push_back("North");
162  }
163  if(FlgUtil::hasFlag(_exit, Direct::EAST)){
164  tmpMenu.push_back("East");
165  }
166  if(FlgUtil::hasFlag(_exit, Direct::SOUTH)){
167  tmpMenu.push_back("South");
168  }
169  if(FlgUtil::hasFlag(_exit, Direct::WEST)){
170  tmpMenu.push_back("West");
171  }
172  if(_end){
173  tmpMenu.push_back("Down");
174  }
175  do{ //While the player hasn't moved Rooms
176  Game::input = menu(tmpMenu, 4); //Display the game menu
177  switch(Game::input){
178  case 'P': //Player Status
179  {
180  Game::player.stat();
181  break;
182  }
183  case 'I': //Inventory
184  {
185  Game::player.selectItm();
186  break;
187  }
188  case 'O': //Options
189  {
190  Flow::gMOpts();
191  if(Game::over){ //If quitting
192  moved = true;
193  }
194  break;
195  }
196  case 'M': //Map
197  {
198  Game::floor.draw(); //Draw map
199  break;
200  }
201  default: //For anything else
202  {
203  if(Game::input == 'N' && FlgUtil::hasFlag(_exit, Direct::NORTH)){ //If moving North
204  Game::floor.move(Game::pos, Direct::NORTH);
205  moved = true;
206  }
207  else if(Game::input == 'E' && FlgUtil::hasFlag(_exit, Direct::EAST)){ //If moving East
208  Game::floor.move(Game::pos, Direct::EAST);
209  moved = true;
210  }
211  else if(Game::input == 'S' && FlgUtil::hasFlag(_exit, Direct::SOUTH)){ //If moving South
212  Game::floor.move(Game::pos, Direct::SOUTH);
213  moved = true;
214  }
215  else if(Game::input == 'W' && FlgUtil::hasFlag(_exit, Direct::WEST)){ //If moving West
216  Game::floor.move(Game::pos, Direct::WEST);
217  moved = true;
218  }
219  else if(Game::input == 'D' && _end){ //If exiting the Floor
220  Game::over = true; //Quit game
221  if(Game::conf.ascArt){ //If ASCII art is on
222  rdTxt("GameData/win.txt"); //Win message
223  }
224  else{ //Otherwise
225  std::cout << "YOU WIN!" << std::endl;
226  }
227  save(); //Save the game
228  moved = true;
229  }
230  }
231  }
232  } while(!moved);
233  }
234 }
