224 std::string Flow::Item::mkDesc(unsigned char elem, ItmType type, unsigned char value){
225  std::stringstream r; //String stream to create a string from
226 
227  if(type == ItmType::Potion){ //If the Item is a potion
228  if(elem == DmgElem::NONE){ //If the Item element is NONE
229  r << "Identifies 1 item. ";
230  }
231  if(elem == DmgElem::ABSOLUT){ //If the Item element is ABSOLUT
232  r << "Fully restores HP and MP. ";
233  }
234  else{ //Otherwise
235  if(FlgUtil::hasFlag(elem, DmgElem::HEALING)){ //If the HEALING flag is on
236  r << "Restores " << toInt(value) << " HP. ";
237  }
238  if(FlgUtil::hasFlag(elem, DmgElem::FIRE)){ //If the FIRE flag is on
239  r << "Increases Attack by " << toInt(value) << ". ";
240  }
241  if(FlgUtil::hasFlag(elem, DmgElem::ICE)){ //If the ICE flag is on
242  r << "Increases Defense by " << toInt(value) << ". ";
243  }
244  if(FlgUtil::hasFlag(elem, DmgElem::LIGHTNG)){ //If the LIGHTNG flag is on
245  r << "Decreases Defense by " << toInt(value) << ". ";
246  }
247  if(FlgUtil::hasFlag(elem, DmgElem::WIND)){ //If the WIND flag is on
248  r << "Decreases Attack by " << toInt(value) << ". ";
249  }
250  if(FlgUtil::hasFlag(elem, DmgElem::HOLY)){ //If the HOLY flag is on
251  r << "Increases max HP by " << toInt(value) << ". ";
252  }
253  if(FlgUtil::hasFlag(elem, DmgElem::SHADOW)){ //If the SHADOW flag is on
254  r << "Increases max MP by " << toInt(value) << ". ";
255  }
256  if(FlgUtil::hasFlag(elem, DmgElem::NGHTMRE)){ //If the NGHTMRE flag is on
257  r << "Obfuscates all items. ";
258  }
259  }
260  }
261  else if(type == ItmType::Weapon){ //If the Item is a weapon
262  if(elem == DmgElem::NONE){ //If no element
263  r << "+" << toInt(value) << " Attack. ";
264  }
265  else{ //Otherwise
266  r << "Deals ( " << DmgElem::toStr(elem) << ") damage. +" << toInt(value) << " Attack. ";
267  }
268  }
269  else if(type == ItmType::Armor){ //Else if the Item is armor
270  if(elem == DmgElem::NONE){ //If no element
271  r << "+" << toInt(value) << " Defense. ";
