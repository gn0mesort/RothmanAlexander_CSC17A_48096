483 void Flow::Actor::attack(Actor &target){
484  int damage = 0; //The total damage to be dealt
485  bool healing = FlgUtil::hasFlag(_weap.element(), DmgElem::HEALING) //Whether or not this attack will heal
486  && _weap.element() != DmgElem::ABSOLUT;
487  bool absorb = false; //Whether or not this attack will be absorbed
488  if(_weap.element() == DmgElem::ABSOLUT && target.armr().element() != DmgElem::ABSOLUT){ //If the attack element is Absolute and the defense element is not Absolute
489  damage = _weap.value() + _atk.value(); //Calculate the damage based solely on attack and weapon damage
490  }
491  else if(target.armr().element() == DmgElem::ABSOLUT){ //Else if both elements are Absolute
492  damage = (_weap.value() + _atk.value()) / ((Game::gmRand.rand() % 3) + 1); //Calculate damage as normal
493  damage -= (target.armr().value() + target.def().value()) / ((Game::gmRand.rand() % 3) + 1);
494  }
495  else if(_weap.element() != DmgElem::ABSOLUT && target.armr().element() == DmgElem::ABSOLUT){ //If the defense element is Absolute but the attack element isn't
496  absorb = true; //The damage is absorbed
497  }
498  else if(_weap.element() == target.armr().element() && _weap.element() != DmgElem::NONE){ //If both elements match and are not None
499  absorb = true; //The damage is absorbed
500  }
501  else{ //Otherwise
502  damage = (_weap.value() + _atk.value()) / ((Game::gmRand.rand() % 3) + 1); //Calculate damage as normal
503  damage -= (target.armr().value() + target.def().value()) / ((Game::gmRand.rand() % 3) + 1);
504  }
505  if(damage < 0){ //If the damage is negative
506  damage = 0; //No damage
507  }
508  if(!healing && !absorb){ //If the attack isn't a heal or absorbed
509  damage += (Game::gmRand.rand() % 5); //Add between 0 and 5 damage
510  if(_weap.element() != DmgElem::ABSOLUT && FlgUtil::hasFlag(_weap.element(), DmgElem::NGHTMRE) && _mp.value() > 0){ //If Nightmare damage
511  std::cout << _name << "'s weapon is burning with witchfire!" << std::endl;
512  int bonus = 0; //The mana bonus is 0
513  bonus = Game::gmRand.rand() % _mp.value(); //Calculate the mana bonus between 0 and the current amount of MP
514  _mp.setValue(_mp.value() - bonus); //Subtract the consumed MP
515  damage += bonus; //Add the bonus to the damage
516  }
517  if(damage > 0){ //If the damage is greater than 0
518  std::cout << _name << " deals " << damage << " damage to " << target.name() << std::endl;
519  target.setHp(target.hp().value() - damage); //Damage the target
520  }
521  else{ //Otherwise
522  std::cout << _name << " missed!" << std::endl; //Missed!
523  }
524  }
525  else if(healing && !absorb){ //If healing
526  std::cout << _name << " heals " << damage << " points to " << target.name() << std::endl;
527  target.setHp(target.hp().value() + damage); //Heal the target
528  }
529  else if(absorb){ //If absorbed
530  std::cout << target.name() << " absorbed the damage!" << std::endl; //No damage
531  }
532

598 void Flow::Actor::toActor(BinArray &data){
599  BinArray strSize(sizeof (unsigned int)), //The size of a string
600  str, //A string
601  job(sizeof (Job)), //The Actor's job
602  hpMax(sizeof (int)), //The Actor's hp
603  mpMax(sizeof (int)), //The Actor's mp

605  def(sizeof (unsigned char)); //The Actor's def
606 
607  data >> strSize; //Extract string size
608  str = BinArray(sizeof (unsigned int) +toInt(strSize)); //Allocate string size
609  data.seekg(data.tellg() - sizeof (unsigned int)); //Seek back to the beginning of the string data
610  data >> str; //Extract string size
611  _name = toStr(str); //Set the Actor's name
612 
613  data >> job; //Extract the job
614  _job = static_cast<Job>(toInt(job)); //Set the Actor's job
