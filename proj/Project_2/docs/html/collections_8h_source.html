345  void LinkedList<T>::remove(unsigned int index){
346  if(_head != 0 && index < _size){ //If the list is not empty and the index is not out of bounds
347  unsigned int count = 0; //The number of nodes deep in the list that the function has gone
348  ListNode<T> *node = _head, //The current node
349  *head = 0, //The last node
350  *tail = 0; //The next node
351  while(node != 0 && count < index){ //While not at the end of the list and not at the selected node
352  node = node->tail(); //The current node is the next node
353  ++count; //increment the count
354  }
355  head = node->head(); //Store the last node
356  tail = node->tail(); //Store the next node
357  if(tail != 0){ //If this node isn't the end of the list
358  node->tail()->head(head); //The next node's head is the last node
359  }
360  else{ //Otherwise
361  _tail = node->head(); //The end of the list is the last node
362  }
363  if(head != 0){ //If this node isn't the beginning of the list
364  node->head()->tail(tail); //The last node's tail is the next node
365  }
366  else{ //Otherwise
367  _head = node->tail(); //The beginning of the list is the next node
368  }
369  --_size; //Reduce the size
370 
371  delete node; //Release the removed node
372  node = 0; //Clear the pointer
373  }
374  else if(_size > 0){ //If the index is out of bounds
