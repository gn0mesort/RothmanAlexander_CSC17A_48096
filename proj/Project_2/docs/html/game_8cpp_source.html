561

566  Job job;
567  int maxHp,
568  maxMp;
569  unsigned char atk,
570  def,

574

576  //Read data from the file

587  in.read(to_bin(armor.pos), sizeof (armor.pos));
588 
589  //Set player data to loaded data
590  player.difficulty(diff);
591  player.job(job);
592  IStat hp = player.hp();
593  hp.max(maxHp);
594  hp.value(hp.max());
595  player.hp(hp);
596  IStat mp = player.mp();
597  mp.max(maxMp);
598  mp.value(mp.max());
599  player.mp(mp);
600  player.attack(atk);
601  player.defense(def);
602  game.player(player);
603 
604  //Regenerate equipped Items from RNGPoints
605  RNGPoint cache;
606  cache.pos = GmRand().pos(); //Cache current RNGPoint
607  cache.seed = GmRand().seed();
608  if(weap.seed != 0){ //Check for default Weapon
609  GmRand().seek(weap);
610  rItem(player, game, ItemType::Weapon, false);
611  player.use(0, false);
612  player.removeItem(0);
613  }
614  else{ //Otherwise provide a default Weapon
615  Weapon weap = player.weapon();
616  weap.name(game.weaponNames()[player.job()][GmRand().rand() % game.weaponNames()[player.job()].size()]);
617  player.equip(weap, false);
618  }
619  if(armor.seed != 0){ //Check for default Armor
620  GmRand().seek(armor);
621  rItem(player, game, ItemType::Armor, false);
622  player.use(0, false);
623  player.removeItem(0);
624  }
625  else{ //Otherwise provide a default Armor
626  player.equip(Armor(DmgElem::NONE, 10), false);
627  }
628  GmRand().seek(cache); //Seek RNG back to cached position
629 
630  game.player(player); //Save the player to the Game object
631  }
632  else{ //Otherwise
633  in.close(); //Close the file
634  throw Error::FileException(); //Throw a FileException
635  }
636  }
637  else{ //Otherwise
638  throw Error::FileException("Invalid File"); //Throw a FileException
639  }
